11 // Extend UIAlertController using a Category to store a code block for
12 // use during dismissal as a property using an associated reference.
13 
14 // This is a workaround for not being able to access UIAlertActions
15 // during programmatic dialog dismissal on iOS 8.
16 
17 // Note that if defined the completion block will be responsible for
18 // the actual dismissal of the dialog to allow for "re-entrant" dialogs
19 // - i.e. the creation of secondary dialogs within the block itself.
20 
21 @interface UIAlertController (VNCExtensions)
22 
23 @property (copy, nonatomic) void (^completion)(UIAlertAction*);
24 
25 @end
